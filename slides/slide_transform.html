ฟังก์ชันการแปลงข้อมูล (Transform function)

1. การเลือกกลยุทธ์ (Strategy Selection)

ตัดสินใจว่าจำเป็นต้องใช้การแมปแบบ value ที่มีความแม่นยำสูงหรือไม่ โดยอิงจากระยะเวลาประวัติของกองทุน

2. จุดยึดทางกายภาพ (Physical Anchors)

คำนวณตำแหน่งพิกเซลที่แน่นอนสำหรับป้ายชื่อ เพื่อข้ามตรรกะการเว้นระยะแบบ "Nice Ticks" มาตรฐานของ ECharts

3. การประมาณค่าในช่วงข้อมูล (Data Interpolation)

แมปจุดข้อมูลแต่ละจุดไปยัง xValue ทางคณิตศาสตร์ เพื่อให้แน่ใจว่าเส้นกราฟจะมีความสมูทแม้จะมีช่องว่างในวันหยุด 


export const transformChartData = (navHistory, timeFrame) => {
  // ขั้นตอนที่ 1: ตรวจสอบบริบทประวัติและเลือกป้ายชื่อ
  const { isMoreThanOneYear } = checkHistory(navHistory);
  const labels = getStrategyLabels(navHistory, timeFrame);

  // ขั้นตอนที่ 2: คำนวณจุดยึดค่า (Value Anchors) หากจำเป็น
  const anchors = getXAxisValueAnchors(labels, navHistory.length - 1);

  const xAxis = {
    // สวิตช์ "เวทมนตร์": บังคับใช้แกนแบบ value เพื่อความแม่นยำ
    type: anchors.length > 0 ? 'value' : 'category',
    axisLabel: { formatter: formatLabel }
  };

  // ขั้นตอนที่ 3: ประมาณค่าในช่วงของจุดข้อมูล
  const data = navHistory.map((item, index) => ({
    ...item,
    ...(anchors.length > 0 && {
      xValue: getInterpolatedXValue(index, anchors)
    }),
  }));

  return { xAxis, data };
};
