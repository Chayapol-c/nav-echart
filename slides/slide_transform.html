Transform function

1. Strategy Selection

Determines if we need high-precision value mapping based on fund history duration.

2. Physical Anchors

Calculates exact pixel positions for labels to bypass ECharts' default "Nice Ticks" spacing logic.

3. Data Interpolation

Maps each data point to a mathematical xValue, ensuring smooth curves even across weekend gaps. 


export const transformChartData = (navHistory, timeFrame) => {
  // Step 1: Detect history context & select labels
  const { isMoreThanOneYear } = checkHistory(navHistory);
  const labels = getStrategyLabels(navHistory, timeFrame);

  // Step 2: Calculate Value Anchors (if needed)
  const anchors = getXAxisValueAnchors(labels, navHistory.length - 1);

  const xAxis = {
    // The "Magic" Switch: Force value axis for precision
    type: anchors.length > 0 ? 'value' : 'category',
    axisLabel: { formatter: formatLabel }
  };

  // Step 3: Interpolate Points
  const data = navHistory.map((item, index) => ({
    ...item,
    ...(anchors.length > 0 && {
      xValue: getInterpolatedXValue(index, anchors)
    }),
  }));

  return { xAxis, data };
};
