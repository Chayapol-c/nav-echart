Distribution

const getMonthGapIndexesWithFallback = (data: IFundHistory[], monthGap: number): number[] => {
  const monthGapIndexes = getMonthGapLabelIndexesFromLatest(data, monthGap);
  const monthBucketIndexes = getMonthBucketLastIndexes(data);
  const expectedCount = Math.min(DEFAULT_COUNT, monthBucketIndexes.length);

  if (monthGapIndexes.length >= expectedCount) {
    return monthGapIndexes;
  }

  // Fallback Logic: Even distribution
  return getEvenlySpacedIndexes(monthBucketIndexes);
};

Step 3.1: Even Distribution Logic

Input

Accepts the array of all valid month-end indices found in Step 2.

Interpolation

Calculate the exact position in the array for each of the 5 labels.

Selection

Use Math.floor to pick the closest index and filter for uniqueness. 


const getEvenlySpacedIndexes = (indexes, labelCount = 5) => {
  if (indexes.length <= labelCount) return indexes;

  const lastIndex = indexes.length - 1;

  return Array.from({ length: labelCount }, (_, pointIndex) => {
    // Formula: (Total Range * Current Step) / (Total Steps)
    const i = Math.floor((lastIndex * pointIndex) / (labelCount - 1));
    return indexes[i];
  }).filter((v, i, self) => self.indexOf(v) === i);
};
