ขั้นตอนที่ 4.1: ตรรกะจุดยึดรายปี

การจัดแนว

ระบุเดือนปัจจุบันเพื่อใช้เป็นจุดยึดที่สอดคล้องกันแบบปีต่อปี

การฉายภาพปี

ลบปีทางคณิตศาสตร์ (T, T-1, T-2) เพื่อระบุเดือนเดียวกัน

↓ กดลงต่อเพื่อดูการนำไปใช้งาน 

const getLatestYearLabelIndexes = (data, labelCount) => {
  const lastDate = dayjs(data[data.length - 1].navDate);
  const targetMonth = lastDate.month();

  return Array.from({ length: labelCount }, (_, pointIndex) => {
    const revIndex = labelCount - pointIndex - 1;
    const year = lastDate.year() - revIndex;
    // เจาะลึก: เราจะหาดัชนีสำหรับปีนี้ได้อย่างไร?
    return findLastIndexByYear(data, year, targetMonth);
  }).filter(v => v !== undefined);
};

ขั้นตอนที่ 4.2: การนำ findLastIndexByYear มาใช้

การค้นหาย้อนกลับ

สแกนข้อมูลย้อนกลับเพื่อหาจุดข้อมูลล่าสุดในปีเป้าหมาย

เดือนที่ตรงกัน

หากพบ targetMonth ให้ส่งคืนทันทีเพื่อการจัดแนวที่สมบูรณ์แบบ

แผนสำรองที่ชาญฉลาด

หากไม่พบเดือนนั้น (วันหยุด) ให้ส่งคืนวันที่ล่าสุดที่มีอยู่ในปีนั้น


const findLastIndexByYear = (data, year, targetMonth) => {
  let fallbackIndex: number | undefined;

  for (let index = data.length - 1; index >= 0; index -= 1) {
    const date = dayjs(data[index].navDate);

    if (date.year() !== year) continue;
    fallbackIndex = fallbackIndex ?? index; // ติดตามรายการล่าสุดที่พบในปีนั้น

    if (date.month() === targetMonth) {
      return index; // พบเดือนที่ตรงกันพอดี!
    }
  }

  return fallbackIndex; // ใช้ข้อมูลล่าสุดที่มีอยู่ในปีนั้น
};
