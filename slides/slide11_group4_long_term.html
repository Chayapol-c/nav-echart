const getLatestYearLabelIndexes = (data, labelCount) => {
  const lastDate = dayjs(data[data.length - 1].navDate);
  const targetMonth = lastDate.month();

  return Array.from({ length: labelCount }, (_, pointIndex) => {
    const revIndex = labelCount - pointIndex - 1;
    const year = lastDate.year() - revIndex;
    // Drill Down: How do we find the index for this year?
    return findLastIndexByYear(data, year, targetMonth);
  }).filter(v => v !== undefined);
};

Step 4.2: findLastIndexByYear Implementation

Reverse Search

Scan data backward to find the latest point in the target year.

Month Match

If we find the targetMonth, return it immediately for perfect alignment.

Smart Fallback

If the month is missing (holiday), return the last available date in that year.


const findLastIndexByYear = (data, year, targetMonth) => {
  let fallbackIndex: number | undefined;

  for (let index = data.length - 1; index >= 0; index -= 1) {
    const date = dayjs(data[index].navDate);

    if (date.year() !== year) continue;
    fallbackIndex = fallbackIndex ?? index; // Track last seen in year

    if (date.month() === targetMonth) {
      return index; // Found perfect match!
    }
  }

  return fallbackIndex; // Use latest available in that year
};
