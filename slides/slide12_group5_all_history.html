Group 5: All

กลยุทธ์: การตรวจจับอัจฉริยะ

"ปรับเปลี่ยนตรรกะแบบไดนามิกตามอายุของกองทุน"

ระบบจะตรวจจับความยาวของประวัติและเลือกอัลกอริทึมที่ดีที่สุด

  • 🕰️ > 1 ปี: ใช้กลุ่มปี (Year Buckets)
  • 📅 > 1 เดือน: ใช้ระยะห่างรายเดือน (Month Gaps)
  • ⚡ < 1 เดือน: ใช้การเว้นระยะทางคณิตศาสตร์ (Math Spacing) 

export const shouldShowXAxisLabelAll = (data, index) => {
  const { isMoreThanOneYear, isMoreThanMonth } = getHistoryDuration(data);

  if (isMoreThanOneYear) {
    // กองทุนที่มีประวัติยาวนาน: ระยะห่างรายปี
    return shouldShowXAxisLabel5Y(index, data);
  }

  if (isMoreThanMonth) {
    // กองทุนที่เปิดมาสักพัก: ระยะห่างรายเดือน
    return shouldShowXAxisLabel1Y(index, data);
  }

  // กองทุนใหม่: การเว้นระยะทางคณิตศาสตร์แบบเท่ากัน
  return shouldShowXAxisLabel1M(data, index);
};
