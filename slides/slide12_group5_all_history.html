Group 5: All

Strategy: Smart Detection

"Dynamically adjust logic based on the fund's age."

The system detects the history length and selects the best algorithm.

  • 🕰️ > 1 Year: Use Year Buckets.
  • 📅 > 1 Month: Use Month Gaps.
  • ⚡ < 1 Month: Use Math Spacing. 

export const shouldShowXAxisLabelAll = (data, index) => {
  const { isMoreThanOneYear, isMoreThanMonth } = getHistoryDuration(data);

  if (isMoreThanOneYear) {
    // Legacy funds: Year intervals
    return shouldShowXAxisLabel5Y(index, data);
  }

  if (isMoreThanMonth) {
    // Established funds: Month gaps
    return shouldShowXAxisLabel1Y(index, data);
  }

  // New funds: Even math spacing
  return shouldShowXAxisLabel1M(data, index);
};
