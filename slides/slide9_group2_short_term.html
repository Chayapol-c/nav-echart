ขั้นตอนที่ 2.1: การนำการคำนวณแบบขอบในมาใช้

การเติมขอบ (Inset)

หลีกเลี่ยงปัญหา "ขอบป้ายชื่อถูกตัด" โดยการเลื่อนป้ายแรกและป้ายสุดท้ายออกจากขอบแผนภูมิ

คณิตศาสตร์อัตราส่วน

คำนวณ positionRatio ระหว่าง inset และ 1-inset

การประมาณค่าในช่วง

"การประมาณค่าเชิงเส้นผ่านดัชนีช่วยให้มั่นใจว่าป้ายชื่อจะอยู่ตรงกลางพอดีกับข้อมูล"


const getInsetEvenlySpacedIndexes = (indexes, labelCount, insetRatio = 0.05) => {
  const lastIndex = indexes.length - 1;

  return Array.from({ length: labelCount }, (_, pointIndex) => {
    // คณิตศาสตร์: ประมาณค่าเชิงเส้นระหว่าง [inset] และ [1 - inset]
    const positionRatio =
      insetRatio + ((1 - insetRatio * 2) * pointIndex) / (labelCount - 1);

    const targetIndex = Math.round(lastIndex * positionRatio);
    return indexes[targetIndex];
  });
};
