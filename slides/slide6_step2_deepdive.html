const getMonthGapIndexesWithFallback = (data: IFundHistory[], monthGap: number): number[] => {

ขั้นตอนที่ 2.1: ตรรกะการจัดกลุ่มวันที่

การจัดกลุ่ม (Grouping)

แมปวันที่เป็นสตริง YYYY-MM เพื่อสร้างกลุ่มที่สมเหตุสมผล

ความไม่ซ้ำกัน (Uniqueness)

ใช้ Map เพื่อให้แน่ใจว่าเก็บเฉพาะดัชนีสุดท้าย (LAST index) ของแต่ละเดือนเท่านั้น

ผลลัพธ์ (Result)

แปลง Map กลับเป็นอาร์เรย์ของดัชนีป้ายชื่อที่ใช้งานได้


const getMonthBucketLastIndexes = (data: IFundHistory[]): number[] =>
  getDateBucketLastIndexes(data, (date) => date.format('YYYY-MM'));
          

const getDateBucketLastIndexes = (data, formatter) => {
  const buckets = new Map();

  data.forEach((item, index) => {
    const key = formatter(dayjs(item.navDate));
    // เขียนทับดัชนีก่อนหน้า เหลือเพียงดัชนีสุดท้ายสำหรับกลุ่มนั้น
    buckets.set(key, index);
  });

  return Array.from(buckets.values());
};
