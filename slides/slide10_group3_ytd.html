Group 3: YTD (ยึดตามปฏิทิน)

กลยุทธ์: จุดสำคัญหลัก (Key Milestones)

"มุ่งเน้นที่จุดเริ่มต้นของปีและจุดกึ่งกลางปี"

  • 📍 มกราคม: จุดเริ่มต้นที่สำคัญ
  • 📍 กึ่งกลาง: จุดกึ่งกลางของความคืบหน้า
  • 📍 ล่าสุด: สถานะปัจจุบัน
↓ กดลงเพื่อดูรายละเอียด 

export const shouldShowXAxisLabelYTD = (index, data) => {
  const latestYear = dayjs(data[data.length - 1].navDate).year();

  // มกราคม: จุดยึดที่สำคัญที่สุดสำหรับ YTD
  const janIndex = findLastIndexByMonthKey(data, `${latestYear}-01`);

  // กึ่งกลาง: จุดกึ่งกลางของข้อมูล YTD ที่มีอยู่
  const midIndex = findLastIndexByMonthKey(data, middleMonthKey);

  // ปัจจุบัน: แสดงถึงราคาล่าสุด
  const lastIndex = data.length - 1;

  const indexes = [janIndex, midIndex, lastIndex];
  return isTargetIndex(indexes, index);
};

Group 3.1: การนำการค้นหาด้วยคีย์เดือนมาใช้

การสร้างคีย์

กำหนดเป้าหมายเฉพาะเดือนโดยใช้รูปแบบ YYYY-MM

การสแกนย้อนกลับ

สแกนจากวันที่ล่าสุดย้อนกลับเพื่อค้นหาจุดข้อมูลสุดท้ายที่มีอยู่ในเดือนนั้น

ประสิทธิภาพ

ส่งคืนผลลัพธ์แรกที่พบทันที เพื่อให้แน่ใจว่าได้ราคา "ปิด" ของเดือนนั้น


const findLastIndexByMonthKey = (data, monthKey) => {
  // เริ่มต้นจากท้ายอาร์เรย์
  for (let index = data.length - 1; index >= 0; index -= 1) {
    const currentKey = dayjs(data[index].navDate).format('YYYY-MM');

    if (currentKey === monthKey) {
      return index; // พบวันซื้อขายวันสุดท้ายของเดือน
    }
  }

  return undefined; // ไม่พบเดือนในข้อมูล
};
