export const shouldShowXAxisLabelYTD = (index, data) => {
  const latestYear = dayjs(data[data.length - 1].navDate).year();

  // January: The most important anchor for YTD
  const janIndex = findLastIndexByMonthKey(data, `${latestYear}-01`);

  // Middle: The midpoint of available YTD data
  const midIndex = findLastIndexByMonthKey(data, middleMonthKey);

  // Current: Always show the latest price
  const lastIndex = data.length - 1;

  const indexes = [janIndex, midIndex, lastIndex];
  return isTargetIndex(indexes, index);
};

Step 3.1: Month Key Lookup Implementation

Key Generation

Target a specific month using the YYYY-MM format.

Backward Scan

Scan from the latest date backward to find the very last available point in that month.

Efficiency

Immediately returns the first match found to ensure we get the "closing" price of that month.


const findLastIndexByMonthKey = (data, monthKey) => {
  // Start from the end of the array
  for (let index = data.length - 1; index >= 0; index -= 1) {
    const currentKey = dayjs(data[index].navDate).format('YYYY-MM');

    if (currentKey === monthKey) {
      return index; // Found the last trading day of the month
    }
  }

  return undefined; // Month not found in data
};
