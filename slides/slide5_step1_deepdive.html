const getMonthGapLabelIndexesFromLatest = (data, monthGap, count = 3) => {
  // ยึดกับจุดข้อมูลล่าสุดเสมอ
  const lastDate = dayjs(data[data.length - 1].navDate);
  const indexes = Array.from({ length: count }, (_, pointIndex) => {
    const revIndex = count - pointIndex - 1;
    const monthKey = lastDate.subtract(revIndex * monthGap, 'month').format('YYYY-MM');
    return findLastIndexByMonthKey(data, monthKey);
  });

  return indexes;
};

const getMonthGapLabelIndexesFromLatest = (data, monthGap, count = 3) => {
  const lastDate = dayjs(data[data.length - 1].navDate);
  const indexes = Array.from({ length: count }, (_, pointIndex) => {
    const revIndex = count - pointIndex - 1;
    const monthKey = lastDate.subtract(revIndex * monthGap, 'month').format('YYYY-MM');
    return findLastIndexByMonthKey(data, monthKey);
  })
  .filter(v => v !== undefined);

  return [...new Set(indexes)]; // ตรวจสอบให้แน่ใจว่าดัชนีไม่ซ้ำกัน
};
